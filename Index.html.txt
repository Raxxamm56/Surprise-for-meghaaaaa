<script>

const data = [
  { img:"photo1.jpg", text:"Meghaaaa Ji ❤️ You are the most beautiful part of my life.", color:"#ff4e50" },
  { img:"photo2.jpg", text:"Every moment with you feels magical 💕", color:"#ff7675" },
  { img:"photo3.jpg", text:"Your smile is my biggest happiness 😊", color:"#fd79a8" },
  { img:"photo4.jpg", text:"I never want to hurt you Meghaaaa Ji 💖", color:"#e84393" },
  { img:"photo5.jpg", text:"You are my forever and always 🌹", color:"#fab1a0" },
  { img:"photo6.jpg", text:"Meghaaaa Ji... I am truly sorry. Please forgive me. You mean everything to me. I promise to love you better, respect you more, and never hurt you again. ❤️", color:"#ff6b81" }
];

let index = 0;

/* Create floating hearts */
setInterval(()=>{
  const heart=document.createElement("div");
  heart.className="heart";
  heart.innerHTML="❤️";
  heart.style.left=Math.random()*100+"%";
  heart.style.fontSize=(15+Math.random()*20)+"px";
  heart.style.animationDuration=(4+Math.random()*4)+"s";
  document.querySelector(".phone").appendChild(heart);

  setTimeout(()=>heart.remove(),7000);
},500);

function start(){
  document.getElementById("intro").classList.add("hidden");
  createBalloon();
}

function createBalloon(){
  if(index>=data.length) return;

  const balloon=document.createElement("div");
  balloon.className="balloon";
  balloon.style.background=data[index].color;
  balloon.style.left=Math.random()*280+"px";

  balloon.onclick=function(){
    showPopup(data[index]);
    balloon.remove();

    if(index===data.length-1){
      document.getElementById("voice").play();
    }

    index++;
    setTimeout(createBalloon,800);
  };

  document.querySelector(".phone").appendChild(balloon);
}

function showPopup(item){
  document.getElementById("popupImage").src=item.img;
  document.getElementById("popupText").innerText=item.text;
  document.getElementById("popup").classList.remove("hidden");
}

function closePopup(){
  document.getElementById("popup").classList.add("hidden");
}

</script>
